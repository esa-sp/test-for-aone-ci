NestJS Example

A progressive Node.js framework for building efficient, scalable server-side apps.

@Controller('api')
export class AppController {

  @Get('health')
  health() {
    return { ok: true, framework: 'nestjs' }
  }

  @Get('users/:id')
  getUser(@Param('id') id: string) {
    return { user: id }
  }
}
